Styling

Font-size

HTML:
<svg>
<textid="font-size-demo"x="50"y="80"font-size="0.5rem">test text</text>
</svg>

CSS:
#font-size-demo{
font-size:0.5rem;
}

Direction

HTML:
<svg>
<textid="direction-demo"x="50%"y="75%"direction="rtl">test text</text>
</svg>

CSS:
#direction-demo{
direction:ltr;
}

Fill

HTML:
<svg>
<rectid="fill-demo"x="25%"y="25%"width="50%"height="50%"fill="black"/>
</svg>

CSS:
#fill-demo{
fill:black;
}

Fill-opacity

HTML:
<svg>
<rectid="fill-opacity-demo"x="25%"y="25%"width="50%"height="50%"fill="black"fill-opacity="50%"/>
</svg>

CSS:
#fill-opacity-demo{
fill-opacity:50%;
}

Color

HTML:
<svg>
<gid="color-demo"color="red">
<rectx="10%"y="25%"width="50px"height="50px"fill="currentcolor"/>
</g>
</svg>

CSS:
#color-demo{
color:red;
}

Writing-mode

HTML:
<svg>
<textid="writing-mode-demo"x="10%"y="10%"writing-mode="tb">test text</text>
</svg>

CSS:
#writing-mode-demo{
writing-mode:tb;
}
