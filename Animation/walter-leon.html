Example 1: Vanishing Square

text here

            
HTML:
<svg xmlns="http://www.w3.org/2000/svg" id="square" width="200px" height="200px">
    <rect height="180px" width="180px" x="10px" y="10px">
        <animate attributeType="CSS"
                 attributeName="opacity"
                 values="1 ; 0 ; 1"
                 keyTimes="0 ; 0.5 ; 1"
                 dur="5s"
                 repeatCount="indefinite"/>
    </rect>
</svg>

CSS:
#square > rect {
    fill: var(--green);
    stroke: var(--darkest);
    stroke-width: 3px;
}

Example 2: Clockhands

text here

            
HTML:
<svg xmlns="http://www.w3.org/2000/svg" id="clock" width="200px" height="200px">

    <!-- The clock -->

    <circle r="40%" cx="50%" cy="50%" fill="white" stroke-width="2%" stroke="black"></circle>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black"></line>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black" transform="rotate(30)"></line>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black" transform="rotate(60)"></line>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black" transform="rotate(90)"></line>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black" transform="rotate(120)"></line>
    <line x1="50%" y1="90%" x2="50%" y2="10%" stroke-width="1%" stroke="black" transform="rotate(150)"></line>
    <circle r="35%" cx="50%" cy="50%" fill="white" stroke-width="0"></circle>
    <circle r="4%" cx="50%" cy="50%" fill="black" stroke-width="0"></circle>

    <!-- The clock hands -->

    <line x1="50%" y1="50%" x2="50%" y2="18%" stroke-width="2%" stroke="black">
        <animateTransform attributeName="transform"
                          attributeType="XML"
                          type="rotate"
                          from="0"
                          to="360"
                          dur="5s"
                          repeatCount="indefinite"/>
    </line>
    <line x1="50%" y1="50%" x2="50%" y2="30%" stroke-width="3%" stroke="black">
        <animateTransform attributeName="transform"
                          attributeType="XML"
                          type="rotate"
                          from="0"
                          to="360"
                          dur="60s"
                          repeatCount="indefinite"/>
    </line>
</svg>

CSS:
#clock > line {
    transform-origin: center;
}

Example 3: Moving Circle

text here

            
HTML:
<svg xmlns="http://www.w3.org/2000/svg" id="circle" width="200px" height="80px">
    <circle id="moving_circle" r="20px" cx="25px" cy="40px"/>

    <!-- The animation is linked with an xlink instead of being nested in the circle -->

    <animate xlink:href="#moving_circle"
             attributeName="cx"
             from="25"
             to="400"
             dur="3.5s"
             repeatCount="1"
             begin="click"
             restart="whenNotActive"/>
</svg>

CSS:
#circle {
    background-color: #FFFFFF4F;
    border: 3px solid var(--darkest);
    border-radius: 1rem;
    margin-top: 1rem;
}

#circle > #moving_circle {
    fill: var(--red);
    stroke: var(--darkest);
    stroke-width: 3px;
}
